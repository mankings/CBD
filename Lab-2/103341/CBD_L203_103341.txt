findSequence= function (seq) {

    var contacts = db.phones.find().toArray();
    var matches = [];

    for (var i = 0; i < contacts.length; i++){
        
        var number = contacts[i].display.split("-")[1];
        if (number.includes(seq)) {
            matches.push(number);
        }

    }

    return matches;
}
